19. Program to Divide a String in 'N' Equal Parts


        // Java Program
        import java.util.Scanner;
        
        public class DivideString {
            public static void main(String[] args) {
                Scanner scanner = new Scanner(System.in);
                System.out.print("Enter a string: ");
                String input = scanner.nextLine();
                System.out.print("Enter number of parts: ");
                int n = scanner.nextInt();
                int length = input.length();
                int partLength = length / n;
        
                for (int i = 0; i < n; i++) {
                    int start = i * partLength;
                    int end = (i == n - 1) ? length : start + partLength; // Last part may take the remainder
                    System.out.println("Part " + (i + 1) + ": " + input.substring(start, end));
                }
            }
        }

Output


        Enter a string: HelloWorld
        Enter number of parts: 5
        Part 1: Hel
        Part 2: loW
        Part 3: orl
        Part 4: d

20. Program to Find All Subsets of a String


        // Java Program
        import java.util.HashSet;
        import java.util.Set;
        
        public class AllSubsets {
            public static void main(String[] args) {
                String str = "abc";
                Set subsets = new HashSet<>();
                int n = str.length();
                for (int i = 0; i < (1 << n); i++) {
                    StringBuilder subset = new StringBuilder();
                    for (int j = 0; j < n; j++) {
                        if ((i & (1 << j)) > 0) {
                            subset.append(str.charAt(j));
                        }
                    }
                    subsets.add(subset.toString());
                }
                System.out.println("All subsets: " + subsets);
            }
        }

Output


        All subsets: [, a, ab, abc, ac, b, bc, c]

21. Program to Find All the Permutations of a String


        // Java Program
        import java.util.ArrayList;
        
        public class Permutations {
            public static void main(String[] args) {
                String str = "abc";
                ArrayList result = new ArrayList<>();
                permute(str, "", result);
                System.out.println("All permutations: " + result);
            }
        
            private static void permute(String str, String prefix, ArrayList result) {
                int n = str.length();
                if (n == 0) {
                    result.add(prefix);
                } else {
                    for (int i = 0; i < n; i++) {
                        permute(str.substring(0, i) + str.substring(i + 1, n), prefix + str.charAt(i), result);
                    }
                }
            }
        }

Output


        All permutations: [abc, acb, bac, bca, cab, cba]

22. Program to Find Maximum and Minimum Occurring Character in a String


        // Java Program
        import java.util.HashMap;
        import java.util.Map;
        
        public class MaxMinOccurringChar {
            public static void main(String[] args) {
                String str = "hello world";
                Map charCount = new HashMap<>();
                
                for (char c : str.toCharArray()) {
                    charCount.put(c, charCount.getOrDefault(c, 0) + 1);
                }
        
                char maxChar = ' ';
                char minChar = ' ';
                int maxCount = Integer.MIN_VALUE;
                int minCount = Integer.MAX_VALUE;
        
                for (Map.Entry entry : charCount.entrySet()) {
                    if (entry.getValue() > maxCount) {
                        maxCount = entry.getValue();
                        maxChar = entry.getKey();
                    }
                    if (entry.getValue() < minCount && entry.getValue() > 0) {
                        minCount = entry.getValue();
                        minChar = entry.getKey();
                    }
                }
        
                System.out.println("Maximum Occurring Character: " + maxChar + " (" + maxCount + ")");
                System.out.println("Minimum Occurring Character: " + minChar + " (" + minCount + ")");
            }
        }

28. Program to Find the Longest Repeating Sequence in a String


        // Java Program
        import java.util.HashMap;
        
        public class LongestRepeatingSequence {
            public static void main(String[] args) {
                String str = "abcabcabc";
                HashMap seqCount = new HashMap<>();
                int maxLength = 0;
                String longestSeq = "";
        
                for (int i = 0; i < str.length(); i++) {
                    for (int j = i + 1; j <= str.length(); j++) {
                        String subStr = str.substring(i, j);
                        seqCount.put(subStr, seqCount.getOrDefault(subStr, 0) + 1);
                        if (seqCount.get(subStr) > 1 && subStr.length() > maxLength) {
                            maxLength = subStr.length();
                            longestSeq = subStr;
                        }
                    }
                }
        
                System.out.println("Longest repeating sequence: " + longestSeq);
            }
        }

Output


        Longest repeating sequence: abc

29. Program to Find the Most Repeated Word in a Text File


        // Java Program
        import java.io.BufferedReader;
        import java.io.FileReader;
        import java.io.IOException;
        import java.util.HashMap;
        
        public class MostRepeatedWord {
            public static void main(String[] args) {
                String filePath = "textfile.txt"; // Specify the path to your text file
                HashMap wordCount = new HashMap<>();
        
                try (BufferedReader br = new BufferedReader(new FileReader(filePath))) {
                    String line;
                    while ((line = br.readLine()) != null) {
                        String[] words = line.toLowerCase().split("\\W+");
                        for (String word : words) {
                            wordCount.put(word, wordCount.getOrDefault(word, 0) + 1);
                        }
                    }
                } catch (IOException e) {
                    e.printStackTrace();
                }
        
                String mostRepeatedWord = "";
                int maxCount = 0;
        
                for (HashMap.Entry entry : wordCount.entrySet()) {
                    if (entry.getValue() > maxCount) {
                        maxCount = entry.getValue();
                        mostRepeatedWord = entry.getKey();
                    }
                }
        
                System.out.println("Most repeated word: " + mostRepeatedWord + " (" + maxCount + ")");
            }
        }

Output


        Most repeated word: the (5) // Example output
