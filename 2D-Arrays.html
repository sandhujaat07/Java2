1. Enter the Matrix of rows and columns entered by the user and print in matrix format


                import java.io.BufferedReader;
                import java.io.InputStreamReader;
                import java.io.IOException;
                
                public class MatrixInput {
                    public static void main(String[] args) throws IOException {
                        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
                
                        System.out.print("Enter number of rows: ");
                        int rows = Integer.parseInt(reader.readLine());
                        System.out.print("Enter number of columns: ");
                        int cols = Integer.parseInt(reader.readLine());
                
                        int[][] matrix = new int[rows][cols];
                
                        System.out.println("Enter the matrix elements:");
                        for (int i = 0; i < rows; i++) {
                            for (int j = 0; j < cols; j++) {
                                matrix[i][j] = Integer.parseInt(reader.readLine());
                            }
                        }
                
                        System.out.println("Matrix:");
                        for (int i = 0; i < rows; i++) {
                            for (int j = 0; j < cols; j++) {
                                System.out.print(matrix[i][j] + " ");
                            }
                            System.out.println();
                        }
                    }
                }

3. Program to Multiply Two Matrices


                import java.io.BufferedReader;
                import java.io.InputStreamReader;
                import java.io.IOException;
                
                public class MatrixMultiplication {
                    public static void main(String[] args) throws IOException {
                        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
                
                        System.out.print("Enter number of rows for the first matrix: ");
                        int rows1 = Integer.parseInt(reader.readLine());
                        System.out.print("Enter number of columns for the first matrix: ");
                        int cols1 = Integer.parseInt(reader.readLine());
                        System.out.print("Enter number of rows for the second matrix: ");
                        int rows2 = Integer.parseInt(reader.readLine());
                        System.out.print("Enter number of columns for the second matrix: ");
                        int cols2 = Integer.parseInt(reader.readLine());
                
                        if (cols1 != rows2) {
                            System.out.println("Matrix multiplication is not possible. Number of columns of the first matrix must be equal to the number of rows of the second matrix.");
                            return;
                        }
                
                        int[][] matrix1 = new int[rows1][cols1];
                        int[][] matrix2 = new int[rows2][cols2];
                        int[][] result = new int[rows1][cols2];
                
                        System.out.println("Enter elements for the first matrix:");
                        for (int i = 0; i < rows1; i++) {
                            for (int j = 0; j < cols1; j++) {
                                matrix1[i][j] = Integer.parseInt(reader.readLine());
                            }
                        }
                
                        System.out.println("Enter elements for the second matrix:");
                        for (int i = 0; i < rows2; i++) {
                            for (int j = 0; j < cols2; j++) {
                                matrix2[i][j] = Integer.parseInt(reader.readLine());
                            }
                        }
                
                        for (int i = 0; i < rows1; i++) {
                            for (int j = 0; j < cols2; j++) {
                                result[i][j] = 0;
                                for (int k = 0; k < cols1; k++) {
                                    result[i][j] += matrix1[i][k] * matrix2[k][j];
                                }
                            }
                        }
                
                        System.out.println("Resultant Matrix after multiplication:");
                        for (int i = 0; i < rows1; i++) {
                            for (int j = 0; j < cols2; j++) {
                                System.out.print(result[i][j] + " ");
                            }
                            System.out.println();
                        }
                    }
                }

5. Program to determine whether two matrices are equal


                import java.io.BufferedReader;
                import java.io.InputStreamReader;
                import java.io.IOException;
                
                public class MatrixEquality {
                    public static void main(String[] args) throws IOException {
                        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
                
                        System.out.print("Enter number of rows: ");
                        int rows = Integer.parseInt(reader.readLine());
                        System.out.print("Enter number of columns: ");
                        int cols = Integer.parseInt(reader.readLine());
                
                        int[][] matrix1 = new int[rows][cols];
                        int[][] matrix2 = new int[rows][cols];
                
                        System.out.println("Enter elements for the first matrix:");
                        for (int i = 0; i < rows; i++) {
                            for (int j = 0; j < cols; j++) {
                                matrix1[i][j] = Integer.parseInt(reader.readLine());
                            }
                        }
                
                        System.out.println("Enter elements for the second matrix:");
                        for (int i = 0; i < rows; i++) {
                            for (int j = 0; j < cols; j++) {
                                matrix2[i][j] = Integer.parseInt(reader.readLine());
                            }
                        }
                
                        boolean isEqual = true;
                        for (int i = 0; i < rows; i++) {
                            for (int j = 0; j < cols; j++) {
                                if (matrix1[i][j] != matrix2[i][j]) {
                                    isEqual = false;
                                    break;
                                }
                            }
                            if (!isEqual) break;
                        }
                
                        if (isEqual) {
                            System.out.println("The matrices are equal.");
                        } else {
                            System.out.println("The matrices are not equal.");
                        }
                    }
                }

8. Program to find the frequency of odd & even numbers in the given matrix


        public class FrequencyOddEven {
            public static void main(String[] args) {
                int[][] matrix = {
                    {1, 2, 3},
                    {4, 5, 6},
                    {7, 8, 9}
                };
        
                int oddCount = 0, evenCount = 0;
        
                for (int[] row : matrix) {
                    for (int num : row) {
                        if (num % 2 == 0) {
                            evenCount++;
                        } else {
                            oddCount++;
                        }
                    }
                }
        
                System.out.println("Odd numbers: " + oddCount);
                System.out.println("Even numbers: " + evenCount);
            }
        }

Output


        Odd numbers: 5
        Even numbers: 4

9. Program to find the sum of each row and each column of a matrix


        public class SumRowsColumns {
            public static void main(String[] args) {
                int[][] matrix = {
                    {1, 2, 3},
                    {4, 5, 6},
                    {7, 8, 9}
                };
        
                System.out.println("Sum of each row:");
                for (int[] row : matrix) {
                    int sum = 0;
                    for (int num : row) {
                        sum += num;
                    }
                    System.out.println(sum);
                }
        
                System.out.println("Sum of each column:");
                for (int j = 0; j < matrix[0].length; j++) {
                    int sum = 0;
                    for (int i = 0; i < matrix.length; i++) {
                        sum += matrix[i][j];
                    }
                    System.out.println(sum);
                }
            }
        }

        15
        24
        Sum of each column:

11. Program to determine whether a given matrix is an identity matrix


        public class IdentityMatrix {
            public static void main(String[] args) {
                int[][] matrix = {
                    {1, 0, 0},
                    {0, 1, 0},
                    {0, 0, 1}
                };
                
                boolean isIdentity = true;
        
                for (int i = 0; i < matrix.length; i++) {
                    for (int j = 0; j < matrix[i].length; j++) {
                        if ((i == j && matrix[i][j] != 1) || (i != j && matrix[i][j] != 0)) {
                            isIdentity = false;
                            break;
                        }
                    }
                }
        
                System.out.println("The matrix is " + (isIdentity ? "an identity matrix." : "not an identity matrix."));
            }
        }

Output


        The matrix is an identity matrix.

12. Program to Transpose matrix


        public class Transpose {
            public static void main(String[] args) {
                int[][] matrix = {
                    {1, 2, 3},
                    {4, 5, 6}
                };
        
                int rows = matrix.length;
                int cols = matrix[0].length;
                int[][] transpose = new int[cols][rows];
        
                for (int i = 0; i < rows; i++) {
                    for (int j = 0; j < cols; j++) {
                        transpose[j][i] = matrix[i][j];
                    }
                }
        
                System.out.println("Transpose of the matrix:");
                for (int[] row : transpose) {
                    for (int num : row) {
                        System.out.print(num + " ");
                    }
                    System.out.println();
                }
            }
        }

Output


        Transpose of the matrix:
        1 4 
        2 5 
        3 6

13. Program to determine whether a given matrix is a sparse matrix


        public class SparseMatrix {
            public static void main(String[] args) {
                int[][] matrix = {
                    {0, 0, 1},
                    {0, 2, 0},
                    {3, 0, 0}
                };
        
                int count = 0;
                for (int[] row : matrix) {
                    for (int num : row) {
                        if (num == 0) {
                            count++;
                        }
                    }
                }
        
                boolean isSparse = count > (matrix.length * matrix[0].length) / 2;
                System.out.println("The matrix is " + (isSparse ? "a sparse matrix." : "not a sparse matrix."));
            }
        }

Output


        The matrix is a sparse matrix.
