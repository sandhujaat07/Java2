1. Java Program to Access java.util.Date Class Through Import Keyword

7. Java Program for Employee Payroll Processing Using Packages


        // File: Emp.java
        package employee;
        
        public class Emp {
            String name;
            int empId;
            String category;
            double bpay, hra, da, npay, pf, grossPay, incomeTax, allowance;
        
            public void calculatePayroll() {
                grossPay = bpay + hra + da + allowance;
                npay = grossPay - (pf + incomeTax);
            }
        }
        
        // File: Emppay.java
        import employee.Emp;
        
        public class Emppay {
            public static void main(String[] args) {
                Emp e = new Emp();
                e.name = "John Doe";
                e.empId = 101;
                e.category = "Developer";
                e.bpay = 50000;
                e.hra = 10000;
                e.da = 5000;
                e.allowance = 2000;
                e.pf = 3000;
                e.incomeTax = 5000;
                e.calculatePayroll();
                System.out.println("Net Pay: " + e.npay);
            }
        }

Output


        Net Pay: 61700.0

8. Java Program to Create a Package with Public Class and Protected Members


        // Create a package named 'protectedpackage'
        package protectedpackage;
        
        public class ProtectedClass {
            protected String message = "Protected Message";
        
            protected void displayMessage() {
                System.out.println(message);
            }
        }
        
        // Another file to access this class
        import protectedpackage.ProtectedClass;
        
        public class Main extends ProtectedClass {
            public static void main(String[] args) {
                Main obj = new Main();
                obj.displayMessage();
            }
        }

Output


        Protected Message
