4. Write a Java program that reads a list of numbers from a file and throws an exception if any of the numbers are positive


        import java.io.File;
        import java.io.FileNotFoundException;
        import java.util.Scanner;
        
        public class PositiveNumberException {
            public static void main(String[] args) {
                try {
                    checkNumbers("numbers.txt");
                } catch (Exception e) {
                    System.out.println("Caught an exception: " + e.getMessage());
                }
            }
        
            static void checkNumbers(String fileName) throws Exception {
                File file = new File(fileName);
                Scanner scanner = new Scanner(file);
                while (scanner.hasNextInt()) {
                    int number = scanner.nextInt();
                    if (number > 0) {
                        throw new Exception("Found a positive number: " + number);
                    }
                }
                scanner.close();
            }
        }

6. Write a Java program that reads a list of integers from the user and throws an exception if any numbers are duplicates


        import java.util.HashSet;
        import java.util.Scanner;
        import java.util.Set;
        
        public class DuplicateNumberException {
            public static void main(String[] args) {
                Scanner scanner = new Scanner(System.in);
                Set numbers = new HashSet<>();
                System.out.println("Enter integers (type 'exit' to finish):");
        
                while (true) {
                    String input = scanner.nextLine();
                    if (input.equals("exit")) break;
        
                    int number = Integer.parseInt(input);
                    if (!numbers.add(number)) {
                        System.out.println("Caught an exception: Duplicate number " + number);
                    }
                }
                scanner.close();
            }
        }

7. Write a Java program to create a method that takes a string as input and throws an exception if the string does not contain vowels

9. Write a Java program to manage the driver with path, username and password, if not successful then throw an exception


        import java.sql.Connection;
        import java.sql.DriverManager;
        import java.sql.SQLException;
        
        public class DriverManagement {
            public static void main(String[] args) {
                try {
                    manageDriver("com.mysql.cj.jdbc.Driver", "jdbc:mysql://localhost:3306/testdb", "root", "password");
                } catch (Exception e) {
                    System.out.println("Caught an exception: " + e.getMessage());
                }
            }
        
            static void manageDriver(String driver, String url, String username, String password) throws Exception {
                try {
                    Class.forName(driver);
                    Connection connection = DriverManager.getConnection(url, username, password);
                    System.out.println("Driver managed successfully.");
                } catch (ClassNotFoundException | SQLException e) {
                    throw new Exception("Driver management failed: " + e.getMessage());
                }
            }
        }

10. Write a Java program to throw the SQL Query, insert, delete, update, if not successful then throw an exception


        import java.sql.Connection;
        import java.sql.DriverManager;
        import java.sql.SQLException;
        import java.sql.Statement;
        
        public class SQLQueryExample {
            public static void main(String[] args) {
                try {
                    executeSQL("INSERT INTO users (name) VALUES ('John Doe')");
                } catch (Exception e) {
                    System.out.println("Caught an exception: " + e.getMessage());
                }
            }
        
            static void executeSQL(String sql) throws Exception {
                String url = "jdbc:mysql://localhost:3306/testdb";
                String username = "root";
                String password = "password";
        
                try (Connection connection = DriverManager.getConnection(url, username, password);
                     Statement statement = connection.createStatement()) {
                    statement.executeUpdate(sql);
                    System.out.println("SQL query executed successfully.");
                } catch (SQLException e) {
                    throw new Exception("SQL query execution failed: " + e.getMessage());
                }
            }
        }
