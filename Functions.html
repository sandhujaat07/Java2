1. Write a Java method to find the smallest number among three numbers


        public class SmallestNumber {
            public static void main(String[] args) {
                System.out.println("Smallest number: " + findSmallest(10, 20, 5));
            }
        
            static int findSmallest(int a, int b, int c) {
                return Math.min(a, Math.min(b, c));
            }
        }

2. Write a Java method to compute the average of three numbers

3. Write a Java method to display the middle character of a string


        public class MiddleCharacter {
            public static void main(String[] args) {
                System.out.println("Middle character(s): " + getMiddleCharacter("example"));
            }
        
            static String getMiddleCharacter(String str) {
                int length = str.length();
                if (length % 2 == 0) {
                    return String.valueOf(str.charAt(length / 2 - 1));
                } else {
                    return str.charAt(length / 2) + "" + str.charAt(length / 2 + 1);
                }
            }
        }

5. Write a Java method to compute the future investment value at a given interest rate for a specified number of years


        public class FutureInvestment {
            public static void main(String[] args) {
                System.out.println("Future value: " + computeFutureValue(1000, 0.05, 10));
            }
        
            static double computeFutureValue(double principal, double interestRate, int years) {
                return principal * Math.pow(1 + interestRate, years);
            }
        }

6. Write a Java method to print characters between two characters (i.e. A to P)


        public class CharacterRange {
            public static void main(String[] args) {
                printCharactersBetween('A', 'P');
            }
        
            static void printCharactersBetween(char start, char end) {
                for (char c = start; c <= end; c++) {
                    System.out.print(c + " ");
                }
            }
        }

7. Write a Java method to check whether a string is a valid password


        public class PasswordValidator {
            public static void main(String[] args) {
                System.out.println("Is valid password: " + isValidPassword("StrongP@ssw0rd"));
            }
        
            static boolean isValidPassword(String password) {
                return password.length() >= 8 && password.matches(".*[A-Z].*") &&
                       password.matches(".*[a-z].*") && password.matches(".*[0-9].*") &&
                       password.matches(".*[!@#$%^&*].*");
            }
        }

9. Write a Java method to find all twin prime numbers less than 100


        public class TwinPrimeFinder {
            public static void main(String[] args) {
                findTwinPrimes(100);
            }
        
            static void findTwinPrimes(int limit) {
                for (int i = 3; i < limit; i++) {
                    if (isPrime(i) && isPrime(i + 2)) {
                        System.out.println("(" + i + ", " + (i + 2) + ")");
                    }
                }
            }
        
            static boolean isPrime(int num) {
                if (num < 2) return false;
                for (int i = 2; i <= Math.sqrt(num); i++) {
                    if (num % i == 0) return false;
                }
                return true;
            }
        }

11. Write a Java method that accepts three integers and checks whether they are consecutive or not. Returns true or false


        public class ConsecutiveChecker {
            public static void main(String[] args) {
                System.out.println("Are consecutive: " + areConsecutive(5, 6, 7));
            }
        
            static boolean areConsecutive(int a, int b, int c) {
                return (a + 1 == b && b + 1 == c) || (b + 1 == a && a + 1 == c) || (c + 1 == a && a + 1 == b);
            }
        }

12. Write a Java method that accepts three integers and returns true if one is the middle point between the other two integers, otherwise false


        public class MiddlePointChecker {
            public static void main(String[] args) {
                System.out.println("Is middle point: " + isMiddlePoint(3, 5, 4));
            }
        
            static boolean isMiddlePoint(int a, int b, int c) {
                return (a + c) / 2 == b || (a + b) / 2 == c || (b + c) / 2 == a;
            }
        }

13. Write a Java method that checks whether all the characters in a given string are vowels (a, e, i, o, u) or not. Return true if each character in the string is a vowel, otherwise return false


        public class VowelChecker {
            public static void main(String[] args) {
                System.out.println("All characters are vowels: " + allVowels("aeiou"));
            }
        
            static boolean allVowels(String str) {
                for (char c : str.toCharArray()) {
                    if ("aeiou".indexOf(c) == -1) {
                        return false;
                    }
                }
                return true;
            }
        }
