3. Java Program to Create a Button and Display Image in the Frame when Clicked


        import javax.swing.*;
        import java.awt.event.*;
        
        public class ButtonImageExample {
            public static void main(String[] args) {
                JFrame frame = new JFrame("Button Image Example");
                JButton button = new JButton("Click to display image");
                
                button.addActionListener(new ActionListener() {
                    public void actionPerformed(ActionEvent e) {
                        JLabel imageLabel = new JLabel(new ImageIcon("path_to_image.jpg"));
                        frame.add(imageLabel);
                        frame.revalidate();
                        frame.repaint();
                    }
                });
                
                frame.add(button);
                frame.setSize(300, 200);
                frame.setVisible(true);
                frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
            }

4. Java Program to Create 2 Radio Buttons and Display Selected Button Label


        import javax.swing.*;
        import java.awt.event.*;
        
        public class RadioButtonExample {
            public static void main(String[] args) {
                JFrame frame = new JFrame("Radio Button Example");
                JRadioButton rb1 = new JRadioButton("Option 1");
                JRadioButton rb2 = new JRadioButton("Option 2");
                ButtonGroup group = new ButtonGroup();
                group.add(rb1);
                group.add(rb2);
                JButton button = new JButton("Check");
        
                button.addActionListener(new ActionListener() {
                    public void actionPerformed(ActionEvent e) {
                        if (rb1.isSelected()) {
                            System.out.println("Option 1 Selected");
                        } else if (rb2.isSelected()) {
                            System.out.println("Option 2 Selected");
                        }
                    }
                });
                
                frame.setLayout(new FlowLayout());
                frame.add(rb1);
                frame.add(rb2);
                frame.add(button);
                
                frame.setSize(300, 200);
                frame.setVisible(true);
                frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
            }

9. Java Program to Create and Switch between Frames using Buttons


        import javax.swing.*;
        import java.awt.event.*;
        
        public class FrameSwitchExample {
            public static void main(String[] args) {
                JFrame frame1 = new JFrame("Frame 1");
                JFrame frame2 = new JFrame("Frame 2");
                JButton button1 = new JButton("Go to Frame 2");
                JButton button2 = new JButton("Go to Frame 1");
        
                button1.addActionListener(e -> {
                    frame1.setVisible(false);
                    frame2.setVisible(true);
                });
                button2.addActionListener(e -> {
                    frame2.setVisible(false);
                    frame1.setVisible(true);
                });
        
                frame1.add(button1);
                frame2.add(button2);
        
                frame1.setSize(300, 200);
                frame2.setSize(300, 200);
        
                frame1.setVisible(true);
                frame1.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
                frame2.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
            }

18. Java Program to Display Several Dots on the Screen Continuously


        import javax.swing.*;
        import java.awt.*;
        
        public class DotsDisplayExample extends JPanel {
            protected void paintComponent(Graphics g) {
                super.paintComponent(g);
                for (int i = 0; i < 100; i++) {
                    int x = (int) (Math.random() * getWidth());
                    int y = (int) (Math.random() * getHeight());
                    g.fillOval(x, y, 5, 5);
                }
            }
        
            public static void main(String[] args) {
                JFrame frame = new JFrame("Dots Display Example");
                frame.add(new DotsDisplayExample());
                frame.setSize(300, 200);
                frame.setVisible(true);
                frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
            }

19. Java Program to Validate the TextField for Only Entering Numbers using KeyListener


        import javax.swing.*;
        import java.awt.event.*;
        
        public class NumberValidationExample {
            public static void main(String[] args) {
                JFrame frame = new JFrame("Number Validation Example");
                JTextField textField = new JTextField(20);
        
                textField.addKeyListener(new KeyAdapter() {
                    public void keyTyped(KeyEvent e) {
                        char c = e.getKeyChar();
                        if (!Character.isDigit(c)) {
                            e.consume();
                        }
                    }
                });
        
                frame.add(textField);
                frame.setSize(300, 200);
                frame.setVisible(true);
                frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
            }

20. Java Program to Validate the TextField for Only Entering Email ID using KeyAdapter


        import javax.swing.*;
        import java.awt.event.*;
        
        public class EmailValidationExample {
            public static void main(String[] args) {
                JFrame frame = new JFrame("Email Validation Example");
                JTextField textField = new JTextField(20);
        
                textField.addKeyListener(new KeyAdapter() {
                    public void keyTyped(KeyEvent e) {
                        char c = e.getKeyChar();
                        if (!Character.isLetterOrDigit(c) && c != '@' && c != '.') {
                            e.consume();
                        }
                    }
                });
        
                frame.add(textField);
                frame.setSize(300, 200);
                frame.setVisible(true);
                frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
            }
